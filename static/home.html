Code Example

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/socket.h>

int main() {
    int server_fd, new_socket;
    struct sockaddr_in address;
    
    // Create socket
    server_fd = socket(AF_INET, SOCK_STREAM, 0);
    
    // Bind and listen
    bind(server_fd, (struct sockaddr *)&address, sizeof(address));
    listen(server_fd, 3);
    
    printf("Server running on port 8080\n");
    return 0;
}
